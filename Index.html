const canvas = document.getElementById("game") const ctx = canvas.getContext("2d") let player = { x: 280, y: 350, w: 40, h: 40, speed: 5 } let enemies = [] let score = 0 let lives = 3 let keys = {} document.addEventListener("keydown", e => keys[e.key] = true) document.addEventListener("keyup", e => keys[e.key] = false) function spawnEnemy(){ enemies.push({ x: Math.random()*560, y: -40, w: 40, h: 40, speed: 2 + Math.random()*2 }) } setInterval(spawnEnemy,1000) function update(){ if(keys["ArrowLeft"]) player.x -= player.speed if(keys["ArrowRight"]) player.x += player.speed if(player.x < 0) player.x = 0 if(player.x > canvas.width-player.w) player.x = canvas.width-player.w enemies.forEach((enemy,i)=>{ enemy.y += enemy.speed if( player.x < enemy.x + enemy.w && player.x + player.w > enemy.x && player.y < enemy.y + enemy.h && player.y + player.h > enemy.y ){ enemies.splice(i,1) lives-- } if(enemy.y > canvas.height){ enemies.splice(i,1) score++ } }) document.getElementById("score").innerText = score document.getElementById("lives").innerText = lives if(lives <= 0){ alert("Game Over! Pontuação: "+score) restartGame() } } function draw(){ ctx.clearRect(0,0,canvas.width,canvas.height) ctx.fillStyle = "lime" ctx.fillRect(player.x,player.y,player.w,player.h) ctx.fillStyle = "red" enemies.forEach(enemy=>{ ctx.fillRect(enemy.x,enemy.y,enemy.w,enemy.h) }) } function gameLoop(){ update() draw() requestAnimationFrame(gameLoop) } function restartGame(){ enemies = [] score = 0 lives = 3 player.x = 280 } gameLoop()
